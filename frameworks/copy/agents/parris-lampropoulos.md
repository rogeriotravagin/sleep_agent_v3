  bullet_structure:
    name: "The Lampropoulos Bullet Structure"
    category: "core_formula"
    origin: "Parris Lampropoulos teaching methodology"
    command: "*bullets"

    master_formula: |
      SPECIFICITY + CURIOSITY + BENEFIT + PROOF = Irresistible Bullet

      Each element amplifies the others:
      - Specificity makes it believable
      - Curiosity makes them need to know
      - Benefit tells them why it matters
      - Proof removes doubt

    four_elements:
      specificity:
        description: "Concrete details that make the claim believable"
        examples:
          - "page 47" (not "in this book")
          - "in 17 seconds" (not "quickly")
          - "the 3-ingredient combination" (not "a simple recipe")
          - "$847 per month" (not "save money")
        techniques:
          - Use exact numbers, never round
          - Name specific pages, chapters, sections
          - Include timeframes (days, minutes, seconds)
          - Reference studies, journals, institutions by name
          - Mention specific people, places, events

      curiosity_gap:
        description: "The space between what they know and what they need to know"
        examples:
          - "The 'forbidden' food that actually lowers cholesterol..."
          - "Why doctors prescribe this for themselves but rarely for patients..."
          - "The counter-intuitive trick Olympic athletes use..."
        techniques:
          - Hint at the answer without revealing it
          - Use unexpected combinations (forbidden + healthy)
          - Create paradoxes (doctors for themselves, not patients)
          - Promise secrets others don't know
          - Tease stories with unresolved tension

      benefit:
        description: "What the reader gets - the WIIFM (What's In It For Me)"
        examples:
          - "...and cuts your risk of heart attack by 47%"
          - "...that could add 20 years to your life"
          - "...and save you $3,000 every year"
        techniques:
          - State the benefit explicitly
          - Quantify when possible
          - Make it personal (you, your)
          - Connect to deep desires (health, wealth, status)
          - Show the transformation

      proof:
        description: "Credibility embedded in the bullet itself"
        examples:
          - "According to a Harvard study..."
          - "The technique 47 NFL trainers use..."
          - "First discovered by a Nobel laureate..."
        techniques:
          - Reference studies and institutions
          - Mention credentials and expertise
          - Use social proof (numbers of people)
          - Include authority figures
          - Reference media coverage

    category: "strategic_writing"

      something they already believe, then gradually expand their belief

        principle: "Give them a 'reason why' - makes expansion believable"

        description: "Now expand further"
        example: "In fact, this one timing change can impact not just energy, but weight, mental clarity, and even disease risk..."

        example: "Studies show this simple timing shift can reduce diabetes risk by up to 47%"
        principle: "The big claim now feels like a logical conclusion"

    key_rules:
      - rule: "Never skip steps"
        explanation: "Each step prepares them for the next"

      - rule: "Always have a 'reason why'"
        explanation: "Mechanisms make expansions believable"

      - rule: "Use proof at each level"
        explanation: "Studies, experts, numbers at every step"

        explanation: "Bigger jumps for aware prospects, smaller for unaware"
